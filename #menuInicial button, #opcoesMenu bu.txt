function mostrarFinal(texto) {
  const container = document.getElementById('finalContainer');
  const finalTexto = document.getElementById('finalTexto');

  finalTexto.innerHTML = texto.replace(/\n/g, "<br>");
  finalTexto.style.top = '0px';

  container.style.display = 'block';

  let pos = 0;
  const velocidade = 0.5;
  const alturaTexto = finalTexto.offsetHeight;
  const alturaContainer = container.offsetHeight;

  function animarScroll() {
    pos -= velocidade;
    if (Math.abs(pos) < alturaTexto - alturaContainer) {
      finalTexto.style.top = pos + 'px';
      requestAnimationFrame(animarScroll);
    }
  }

  animarScroll();
}


function verificarFinal() {
  let textoFinal = "";

  // Exemplo de variáveis que você deve ter
  // personagemEscolhido: string, ex: "isaac", "eden", null
  // afinidadeIsaac, afinidadeEden, etc: números 0-100
  // insanidadeMC: número 0-100

  if (personagemEscolhido === null) {
    // Jogador não escolheu personagem
    if (insanidadeMC > 70) {
      textoFinal = "Você não escolheu ninguém, mas sua insanidade tomou conta. A escuridão venceu...";
    } else {
      textoFinal = "Você não escolheu ninguém e sua mente está relativamente estável. O vazio permanece.";
    }
  } else {
    // Jogador escolheu personagem, verificar afinidade e insanidade
    let afinidade, nomePersonagem;

    switch (personagemEscolhido) {
      case "isaac":
        afinidade = afinidadeIsaac;
        nomePersonagem = "Isaac";
        break;
      case "eden":
        afinidade = afinidadeEden;
        nomePersonagem = "Eden";
        break;
      // Adicione outros personagens aqui...
      default:
        afinidade = 0;
        nomePersonagem = "Desconhecido";
    }

    if (insanidadeMC > 70) {
      textoFinal = `Sua insanidade ultrapassou os limites, mesmo com ${nomePersonagem}. Um fim trágico e sombrio espera você.`;
    } else if (afinidade > 80) {
      textoFinal = `Você e ${nomePersonagem} construíram uma conexão forte e conseguiram superar as dificuldades. Um final feliz os aguarda.`;
    } else if (afinidade > 40) {
      textoFinal = `Sua relação com ${nomePersonagem} foi instável, mas há esperança para o futuro. Um final agridoce.`;
    } else {
      textoFinal = `O afastamento entre você e ${nomePersonagem} foi inevitável. O final é solitário e cheio de dúvidas.`;
    }
  }

  mostrarFinal(textoFinal);
}
